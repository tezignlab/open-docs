你需要将素材上传组件的链接用 iframe 嵌入你的系统，你的代码和素材上传组件会基于[postMessage](https://developer.mozilla.org/en-US/docs/Web/API/Window/postMessage)来进行通信交互数据。

```html
<!DOCTYPE html>
<html lang="zh-CN">
  <head>
    <meta charset="UTF-8" />
    <title>素材选择器父页面</title>
  </head>
  <body>
    <iframe
      src="https://xxx.tezign.com/dam_enterprise/material_selector"
      id="iframe"
    />
  </body>
  <script>
    const receiveMessage = (event) => {
      if (event?.data?.type === 'tezign-selector-page-ready') {
        // iframe 页面加载完成，可以传入配置参数
        postMessage();
        return;
      }
      if (event?.data?.type === 'tezign-selector-confirm-btn') {
        // 点击【确认】 按钮 -- 这里接收已选择的素材集合
        // todo
        console.log(event.data.selectedMaterial);
        return;
      }
      if (event?.data?.type === 'tezign-selector-cancel-btn') {
        // 点击【取消】 按钮
        // todo
        console.log(event.data);
        return;
      }
    };
    const postMessage = () => {
      //素材选择器
      const iframe = document.getElementById('iframe');

      iframe.contentWindow.postMessage(
        {
          type: 'tezign-material-selector',
          data: {
            config: [
              { id: 'image', limit: 5 },
              { id: 'audio', limit: 9 },
            ],
            filterCode: 'open-component-search-001',
            formatLimit: 5,
          },
        },
        '*',
      );
    };
    window.addEventListener('message', receiveMessage, false);
  </script>
</html>
```

```typescript jsx
import React, { useEffect } from 'react';

export default function App() {
  //获取素材选择信息
  useEffect(() => {
    window.addEventListener('message', receiveMessage, false);
    return () => {
      window.removeEventListener('message', receiveMessage);
    };
  }, []);

  const receiveMessage = (event) => {
    if (event?.data?.type === 'tezign-selector-page-ready') {
      // iframe 页面加载完成，可以传入配置参数
      postMessage();
      return;
    }
    if (event?.data?.type === 'tezign-selector-confirm-btn') {
      // 点击【确认】 按钮 -- 这里接收已选择的素材集合
      // todo
      console.log(event.data.selectedMaterial);
      return;
    }
    if (event?.data?.type === 'tezign-selector-cancel-btn') {
      // 点击【取消】 按钮
      // todo
      console.log(event.data);
      return;
    }
  };

  const postMessage = () => {
    //素材选择器
    const iframe = document.getElementById('iframe') as HTMLIFrameElement;

    iframe.contentWindow.postMessage(
      {
        type: 'tezign-material-selector',
        data: {
          config: [
            { id: 'image', limit: 5 },
            { id: 'audio', limit: 9 },
          ],
          filterCode: 'open-component-search-001',
          formatLimit: 5,
        },
      },
      '*',
    );
  };

  // 此iframeUrl 填入素材选择器链接
  const iframeUrl = 'https://xxx.tezign.com/dam_enterprise/material_selector';
  return (
    <div>
      <iframe src={iframeUrl} id="iframe" />
    </div>
  );
}
```
